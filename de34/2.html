コード

  1. #include <Adafruit_NeoPixel.h>
  2. #define PIN 9 // 信号用のピンを指定
  3. #define NUMPIXELS 5 // LEDの数を指定
  4. int brightness=50;//明るさ
  6. Adafruit_NeoPixel pixels(NUMPIXELS, PIN, NEO_GRB + NEO_KHZ800);
  8. void setup() {
  9.   pixels.begin(); // NeoPixel出力ピンの初期化
  10.   pixels.setBrightness(brightness);
  11. }
  13. void loop() {
  14.   pixels.clear(); // すべてのLEDの色を0にセット
  16.   for(int i=0; i<NUMPIXELS; i++) {
  17.     int r=random(0,255);
  18.     int g=random(0,255)
  19.     int b=random(0,255)
  20.     
  21.     
  22.     pixels.setPixelColor(i, pixels.Color(r, g, b));
  23.     pixels.show();
  24.     delay(1000);
  25.   }
  26. }

説明

・日常的に感じているような感情やシチュエーションが多かった。

整うとは

整うという言葉を使うときは、銭湯へ行きサウナののちに水風呂、そして外気浴をしたときについこぼれ出る言葉である。
具体的な状況としては、・頭がボーっとする　・リラックスしている　・幸福感がある
整いたいと思うときは、疲れているとき、タスクを終えたとき、など何かを達成したときにそれをより強めたくなったときに行く傾向がある
